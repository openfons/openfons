def ffmpeg_exists() -> bool:

def run_ffmpeg(args: list[str], timeout_seconds: int = 1800) -> tuple[bool, str]:
    try:
        completed = subprocess.run(args, capture_output=True, text=True, timeout=timeout_seconds, check=False)
        if completed.returncode == 0:
            return True, ""
        return False, completed.stderr or completed.stdout or f"exit={completed.returncode}"
    except Exception as exc:
        return False, str(exc)

    # Deterministic placeholder image artifact for compatibility-only phase.
    payload = f"overlay {width}x{height} {' | '.join(lines)}".encode("utf-8")
    Path(output_path).write_bytes(payload)
