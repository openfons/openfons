def wait_for_openclaw_result(
    api_base: str,
    task_id: str,
    *,
    api_token: str = "",
    timeout_seconds: float = 1800.0,
    poll_interval_seconds: float = 5.0,
    request_timeout_seconds: float = 15.0,
    session: Any = None,
) -> dict[str, Any]:
    deadline = time.time() + max(1.0, float(timeout_seconds))
    while True:
        payload = fetch_openclaw_result(
            api_base=api_base,
            task_id=task_id,
            api_token=api_token,
            timeout_seconds=request_timeout_seconds,
            session=session,
        )
        result = payload.get("openclaw_result") if isinstance(payload.get("openclaw_result"), dict) else {}
        status = str(result.get("status") or "").strip().lower()
        if status in {"completed", "failed", "cancelled"}:
            return payload
        if time.time() >= deadline:
            raise TimeoutError(f"wait openclaw result timed out: task_id={task_id} status={status or 'unknown'}")
        time.sleep(max(0.2, float(poll_interval_seconds)))
