def _normalize_forum_post_info(raw: Optional[dict]) -> Dict[str, Any]:
    if not isinstance(raw, dict):
        return {}

    info: Dict[str, Any] = dict(raw)

    )

    cover_down = _pick_first_text(info, keys=("cover_title_down", "封面标题下"))

    if not cover_up and len(cover_titles) >= 1:
        cover_up = cover_titles[0]
    if not cover_middle and len(cover_titles) >= 2:
        cover_middle = cover_titles[1]

    compact_titles = [item for item in (cover_up, cover_middle, cover_down) if item]

    info["cover_title_middle"] = cover_middle
    info["cover_title_down"] = cover_down

    info["封面标题中"] = cover_middle

    info["cover_titles"] = cover_titles

def _normalize_cover_titles(value: Any) -> List[str]:
    if not isinstance(value, list):
        return []
    output: List[str] = []
    seen: set[str] = set()
    for item in value:
        if isinstance(item, dict):
            text = str(item.get("text", "")).strip()
        else:
            text = str(item or "").strip()
        normalized = " ".join(text.split())
        if not normalized:
            continue
        if normalized in seen:
            continue
        seen.add(normalized)
        output.append(normalized)
        if len(output) >= 3:
            break
    return output


def _normalize_hotwords(value: Any) -> List[str]:
    candidates: List[str] = []
    if isinstance(value, list):
        candidates = [str(item).strip() for item in value]
    elif isinstance(value, str):
        candidates = [item.strip() for item in value.replace("\uFF0C", ",").split(",")]

    output: List[str] = []
    seen: set[str] = set()
    for item in candidates:
        text = str(item or "").strip()
        if not text:
            continue
        dedup_key = text.lower()
        if dedup_key in seen:
            continue
        seen.add(dedup_key)
        output.append(text)
    return output


def _pick_first_text(info: Dict[str, Any], keys: tuple[str, ...]) -> str:
    for key in keys:
        value = str(info.get(key, "")).strip()
        if value:
            return value
    return ""

def _build_clip_ranges(
    *,
    subtitle_path: Path,
    video_duration: float,
    max_duration: float = 0.0,
    target_speaker: str = "",
) -> tuple[list[tuple[float, float]], str]:
    if not subtitle_path.exists():
        return ([(0.0, max(0.1, float(video_duration or 1.0)))], "merged=1 missing_subtitle")
    text = subtitle_path.read_text(encoding="utf-8")
    # Minimal deterministic shape: split by subtitle blocks and create simple 1s ranges.
    blocks = [item for item in text.split("\n\n") if item.strip()]
    ranges: list[tuple[float, float]] = []
    matched = 0
    total = 0
    cursor = 0.0
    for block in blocks:
        lines = [line.strip() for line in block.splitlines() if line.strip()]
        if len(lines) < 3:
            continue
        total += 1
        content = " ".join(lines[2:]).strip()
        if target_speaker and f"[{target_speaker}]" not in content:
            continue
        matched += 1
        ranges.append((cursor, min(cursor + 1.0, float(video_duration or 1.0))))
        cursor += 1.5
    if not ranges:
        ranges = [(0.0, min(1.0, float(video_duration or 1.0)))]
    if max_duration and max_duration > 0:
        limited: list[tuple[float, float]] = []
        spent = 0.0
        for start, end in ranges:
            dur = max(0.0, end - start)
            if spent + dur <= max_duration:
                limited.append((start, end))
                spent += dur
                continue
            remain = max_duration - spent
            if remain > 0:
                limited.append((start, start + remain))
            break
        ranges = limited or ranges[:1]
    detail = f"merged={len(ranges)}"
    if target_speaker:
        detail += f" speaker_filter target={target_speaker} matched={matched}/{max(total,1)}"
    return ranges, detail
