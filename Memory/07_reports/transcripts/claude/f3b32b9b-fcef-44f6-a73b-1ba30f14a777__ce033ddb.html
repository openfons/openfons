   320→/** Extract role/timestamp/text from clean markdown format. */
   321→export function parseMessagesFromMarkdown(content: string): RenderedMessage[] {
   322→  const messages: RenderedMessage[] = []
   323→
   324→  // Find ## Messages section
   325→  const msgSection = content.indexOf('\n## Messages')
   326→  if (msgSection === -1) return []
   327→
   328→  const body = content.slice(msgSection + 12) // after "## Messages\n"
   329→
   330→  // Split on ### N. role headers
   331→  const msgRe = /^### \d+\.\s+(\w+)/gm
   332→  let match: RegExpExecArray | null
   333→  const starts: Array<{ index: number; role: string }> = []
   334→
   335→  while ((match = msgRe.exec(body)) !== null) {
   336→    starts.push({ index: match.index, role: match[1] ?? 'unknown' })
   337→  }
   338→
   339→  for (let i = 0; i < starts.length; i++) {
   340→    const start = starts[i]
   341→    if (!start) continue
   342→    const nextStart = starts[i + 1]
   343→    const chunk = nextStart
   344→      ? body.slice(start.index, nextStart.index)
   345→      : body.slice(start.index)
   346→
   347→    const lines = chunk.split('\n')
   348→    // First line is the header, second may be timestamp
   349→    let timestamp = ''
   350→    const textLines: string[] = []
   351→    let pastHeader = false
   352→
   353→    for (const line of lines) {
   354→      if (!pastHeader && line.startsWith('### ')) {
   355→        pastHeader = true
   356→        continue
   357→      }
   358→      if (!pastHeader) continue
   359→      const tsMatch = line.match(/^- Timestamp:\s*`([^`]+)`/)
   360→      if (tsMatch && !timestamp) {
   361→        timestamp = tsMatch[1] ?? ''
   362→        continue
   363→      }
   364→      textLines.push(line)
   365→    }
   366→
   367→    // Remove leading/trailing blank lines from text
   368→    const text = textLines.join('\n').trim()
   369→    if (text || timestamp) {
   370→      messages.push({ role: start.role, timestamp, text })
   371→    }
   372→  }
   373→
   374→  return messages
   375→}
