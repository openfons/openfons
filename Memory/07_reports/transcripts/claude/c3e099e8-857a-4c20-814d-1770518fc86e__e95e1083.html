export interface VectorAdapter {
  upsert(id: string, embedding: number[], metadata: VectorMetadata): void
  search(embedding: number[], topK: number): VectorResult[]
  delete(id: string): void
  close(): void
}

Key decisions:
- embedding is number[] — simple, no external dependency
- VectorMetadata is a flat string-keyed record — sufficient for Phase 1 and compatible with Qdrant's payload model
- All methods synchronous for consistency with the SQLite pattern; async can be added in task 1.5 when Qdrant is wired

## Step 4: Implement SqliteFtsAdapter

File: packages/core/src/storage/search/sqlite-fts-adapter.ts

This is a thin wrapper around the three existing functions in transcript/db.ts. It holds the dbPath as state and delegates all calls.

import type { ManifestEntry } from '@openmnemo/types'
import type { SearchResult } from '../../transcript/db.js'
import { upsertSearchIndex, searchTranscripts } from '../../transcript/db.js'
import type { SearchAdapter, SearchOptions } from './search-adapter.js'

export class SqliteFtsAdapter implements SearchAdapter {
  constructor(private readonly dbPath: string) {}

  search(query: string, options: SearchOptions = {}): SearchResult[] {
    return searchTranscripts(this.dbPath, query, options.limit ?? 20)
  }

  upsert(manifest: ManifestEntry): void {
    upsertSearchIndex(this.dbPath, manifest)
  }

  close(): void {
    // stateless — db is opened/closed per-call in db.ts
  }
}

Key decisions:
- No state beyond dbPath — db.ts already opens and closes the connection per call, so close() is genuinely a no-op
- Default limit of 20 matches the existing default in searchTranscripts
- No logic duplication — all SQL stays in db.ts

## Step 7: factory.ts

File: packages/core/src/storage/factory.ts

The factory takes a minimal config shape (not the full sync Config to avoid a cross-package dependency) and returns adapter instances.

import { join } from 'node:path'
import { SqliteFtsAdapter } from './search/sqlite-fts-adapter.js'
import { QdrantVectorAdapter } from './vector/qdrant-adapter.js'
import { Neo4jGraphAdapter } from './graph/neo4j-adapter.js'
import type { SearchAdapter } from './search/search-adapter.js'
import type { VectorAdapter } from './vector/vector-adapter.js'
import type { GraphAdapter } from './graph/graph-adapter.js'

export interface StorageConfig {
  indexDir: string                          // e.g. ~/.memorytree/index
  vector_backend?: 'sqlite-vec' | 'qdrant'
  graph_backend?: 'sqlite' | 'neo4j'
  qdrant_url?: string
  qdrant_collection?: string
  neo4j_uri?: string
  neo4j_user?: string
  neo4j_password?: string
}

export function createSearchAdapter(cfg: StorageConfig): SearchAdapter {
  const dbPath = join(cfg.indexDir, 'search.sqlite')
  return new SqliteFtsAdapter(dbPath)
}

export function createVectorAdapter(cfg: StorageConfig): VectorAdapter {
  if (cfg.vector_backend === 'qdrant') {
    return new QdrantVectorAdapter(
      cfg.qdrant_url ?? 'http://localhost:6333',
      cfg.qdrant_collection ?? 'transcripts',
    )
  }
  throw new Error('sqlite-vec VectorAdapter not implemented in Phase 1')
}

export function createGraphAdapter(cfg: StorageConfig): GraphAdapter {
  if (cfg.graph_backend === 'neo4j') {
    return new Neo4jGraphAdapter(
      cfg.neo4j_uri ?? 'bolt://localhost:7687',
      cfg.neo4j_user ?? 'neo4j',
      cfg.neo4j_password ?? '',
    )
  }
  throw new Error('sqlite GraphAdapter not implemented in Phase 1')
}

Key decisions:
- StorageConfig is defined locally in factory.ts, not imported from @openmnemo/sync — avoids a circular package dependency
- createVectorAdapter and createGraphAdapter throw for Phase 1 backends (sqlite-vec, sqlite) since those implementations don't exist yet — this is intentional and honest
- createSearchAdapter is the only factory function that works end-to-end in Phase 1
