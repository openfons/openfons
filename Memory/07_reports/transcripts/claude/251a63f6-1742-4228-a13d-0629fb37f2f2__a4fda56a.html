   313→/** Extract role/timestamp/text from clean markdown format. */
   314→export function parseMessagesFromMarkdown(content: string): RenderedMessage[] {
   315→  const messages: RenderedMessage[] = []
   316→
   317→  // Find ## Messages section
   318→  const msgSection = content.indexOf('\n## Messages')
   319→  if (msgSection === -1) return []
   320→
   321→  const body = content.slice(msgSection + 12) // after "## Messages\n"
   322→
   323→  // Split on ### N. role headers
   324→  const msgRe = /^### \d+\.\s+(\w+)/gm
   325→  let match: RegExpExecArray | null
   326→  const starts: Array<{ index: number; role: string }> = []
   327→
   328→  while ((match = msgRe.exec(body)) !== null) {
   329→    starts.push({ index: match.index, role: match[1] ?? 'unknown' })
   330→  }
   331→
   332→  for (let i = 0; i < starts.length; i++) {
   333→    const start = starts[i]
   334→    if (!start) continue
   335→    const nextStart = starts[i + 1]
   336→    const chunk = nextStart
   337→      ? body.slice(start.index, nextStart.index)
   338→      : body.slice(start.index)
   339→
   340→    const lines = chunk.split('\n')
   341→    // First line is the header, second may be timestamp
   342→    let timestamp = ''
   343→    const textLines: string[] = []
   344→    let pastHeader = false
   345→
   346→    for (const line of lines) {
   347→      if (!pastHeader && line.startsWith('### ')) {
   348→        pastHeader = true
   349→        continue
   350→      }
   351→      if (!pastHeader) continue
   352→      const tsMatch = line.match(/^- Timestamp:\s*`([^`]+)`/)
   353→      if (tsMatch && !timestamp) {
   354→        timestamp = tsMatch[1] ?? ''
   355→        continue
   356→      }
   357→      textLines.push(line)
   358→    }
   359→
   360→    // Remove leading/trailing blank lines from text
   361→    const text = textLines.join('\n').trim()
   362→    if (text || timestamp) {
   363→      messages.push({ role: start.role, timestamp, text })
   364→    }
   365→  }
   366→
   367→  return messages
   368→}
