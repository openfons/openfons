   229→                continue
   230→            if not isinstance(block, dict):
   231→                continue
   232→            block_type = str(block.get("type") or "").lower()
   233→            if block_type in TEXT_BLOCK_TYPES:
   234→                text = str(block.get("text") or "").strip()
   235→                if text:
   236→                    text_parts.append(text)
   237→                continue
   238→            if block_type in SKIP_BLOCK_TYPES:
   239→                continue
   240→            if block_type in TOOL_USE_TYPES:
   241→                name = str(block.get("name") or "tool_use")
   242→                tool_events.append(
   243→                    TranscriptToolEvent(
   244→                        summary=f"{name} input={summarize_value(block.get('input'))}",
   245→                        timestamp=timestamp,
   246→                    )
   247→                )
   248→                continue
   249→            if block_type in TOOL_RESULT_TYPES:
   250→                tool_use_id = str(block.get("tool_use_id") or "tool_result")
   251→                tool_events.append(
   252→                    TranscriptToolEvent(
   253→                        summary=f"{tool_use_id} output={summarize_value(block.get('content'))}",
   254→                        timestamp=timestamp,
   255→                    )
   256→                )
   257→
   258→        text = join_paragraphs(text_parts)
   259→        if role in {"user", "assistant"} and text:
   260→            messages.append(TranscriptMessage(role=role, text=text, timestamp=timestamp))
   261→
   262→    return ParsedTranscript(
   263→        client="claude",
   264→        session_id=session_id,
   265→        title=title,
   266→        started_at=started_at,
   267→        cwd=cwd,
   268→        branch=branch,
   269→        messages=messages,
   270→        tool_events=tool_events,
   271→        source_path=path,
   272→    )

   275→def parse_gemini_transcript(path: Path) -> ParsedTranscript:
   276→    session_id = path.stem
   277→    title = path.stem
   278→    started_at = normalize_timestamp(path.stat().st_mtime)
   279→    cwd = ""
   280→    branch = ""
   281→    messages: list[TranscriptMessage] = []
   282→    tool_events: list[TranscriptToolEvent] = []
   283→
   284→    if path.suffix.lower() == ".jsonl":
   285→        payload: Any = load_jsonl(path)
   286→    else:
   287→        payload = json.loads(path.read_text(encoding="utf-8"))
   288→
   289→    message_signatures: set[tuple[str, str, str | None]] = set()
   290→    first_meta = find_first_mapping_with_keys(payload, {"sessionId", "chatId", "cwd", "branch", "timestamp"})
   291→    if first_meta:
   292→        session_id = str(first_meta.get("sessionId") or first_meta.get("chatId") or first_meta.get("id") or session_id)
   293→        started_at = earliest_timestamp(started_at, first_meta.get("timestamp"))
   294→        cwd = str(first_meta.get("cwd") or first_meta.get("projectRoot") or cwd)
   295→        branch = str(first_meta.get("branch") or branch)
   296→
   297→    def visit(node: Any) -> None:
   298→        nonlocal session_id, started_at, cwd, branch
   299→        if isinstance(node, list):
   300→            for item in node:
   301→                visit(item)
   302→            return
   303→        if not isinstance(node, dict):
   304→            return
   305→
   306→        session_id = str(node.get("sessionId") or node.get("chatId") or node.get("id") or session_id)
   307→        started_at = earliest_timestamp(started_at, node.get("timestamp"))
   308→        cwd = str(node.get("cwd") or node.get("projectRoot") or cwd)
   309→        branch = str(node.get("branch") or branch)
   310→
   311→        role = str(node.get("role") or node.get("author") or node.get("sender") or "").lower()
   312→        timestamp = normalize_timestamp(node.get("timestamp"), started_at)
   313→        if role in {"user", "assistant", "model"}:
   314→            normalized_role = "assistant" if role == "model" else role
   315→            text = extract_gemini_text(node)
   316→            signature = (normalized_role, text, timestamp)
   317→            if text and signature not in message_signatures:
   318→                message_signatures.add(signature)
   319→                messages.append(TranscriptMessage(role=normalized_role, text=text, timestamp=timestamp))
   320→
   321→        tool_name = (
   322→            get_nested(node, "toolUse", "name")
   323→            or node.get("toolName")
   324→            or node.get("functionName")
   325→            or node.get("tool")
   326→        )
   327→        if tool_name:
   328→            tool_events.append(
   329→                TranscriptToolEvent(
   330→                    summary=f"{tool_name} input={summarize_value(node.get('args') or node.get('input'))}",
   331→                    timestamp=timestamp,
   332→                )
   333→            )
   334→
   335→        for value in node.values():
   336→            visit(value)
   337→
   338→    visit(payload)
   339→
   340→    return ParsedTranscript(
   341→        client="gemini",
   342→        session_id=session_id,
   343→        title=title,
   344→        started_at=started_at,
   345→        cwd=cwd,
   346→        branch=branch,
   347→        messages=messages,
   348→        tool_events=tool_events,
   349→        source_path=path,
   350→    )

   354→    parsed: ParsedTranscript,

   381→        global_raw_path,
   382→        global_clean_path,
   383→        global_manifest_path,
   384→        global_event_log_path,
   385→        global_db_path,

   387→        path.parent.mkdir(parents=True, exist_ok=True)
   388→    if mirror_to_repo:
   389→        for path in (repo_raw_path, repo_clean_path, repo_manifest_path):
   390→            path.parent.mkdir(parents=True, exist_ok=True)
   391→
   392→    if mirror_to_repo:
   393→        copy_file(parsed.source_path, repo_raw_path)
   394→    copy_file(parsed.source_path, global_raw_path)
   395→
   396→    manifest = {
   397→        "client": parsed.client,
   398→        "project": project_slug,
   399→        "session_id": parsed.session_id,
   400→        "title": parsed.title,
   401→        "started_at": parsed.started_at,

   408→        "repo_raw_path": repo_raw_path.relative_to(root).as_posix() if mirror_to_repo else "",
   409→        "repo_clean_path": repo_clean_path.relative_to(root).as_posix() if mirror_to_repo else "",
   410→        "repo_manifest_path": repo_manifest_path.relative_to(root).as_posix() if mirror_to_repo else "",
   411→        "global_raw_path": global_raw_path.as_posix(),
   412→        "global_clean_path": global_clean_path.as_posix(),
   413→        "global_manifest_path": global_manifest_path.as_posix(),
   414→        "message_count": len(parsed.messages),
   415→        "tool_event_count": len(parsed.tool_events),
   416→        "cleaning_mode": "deterministic-code",
   417→        "repo_mirror_enabled": mirror_to_repo,
   418→    }
   419→    existing_global_manifest = load_json(global_manifest_path)
   420→    manifest = preserve_existing_import_timestamp(existing_global_manifest, manifest)
   421→
   422→    if mirror_to_repo:
   423→        write_clean_markdown(parsed, manifest, repo_clean_path)
   424→    write_clean_markdown(parsed, manifest, global_clean_path)
   425→    if mirror_to_repo:
   426→        write_json(repo_manifest_path, manifest)
   427→    append_to_event_log = existing_global_manifest is None or existing_global_manifest != manifest
   428→    write_json(global_manifest_path, manifest)
   429→    if append_to_event_log:
   430→        append_jsonl(global_event_log_path, manifest)
   431→    upsert_search_index(global_db_path, manifest)
   432→
   433→    result = dict(manifest)
   434→    result["global_index_db"] = global_db_path.as_posix()
   435→    return result

   679→def extract_text_blocks(blocks: Any) -> str:
   680→    parts: list[str] = []
   681→    for block in ensure_list(blocks):
   682→        if isinstance(block, str):
   683→            text = block.strip()
   684→            if text:
   685→                parts.append(text)
   686→            continue
   687→        if not isinstance(block, dict):
   688→            continue
   689→        block_type = str(block.get("type") or "").lower()
   690→        if block_type in SKIP_BLOCK_TYPES | TOOL_USE_TYPES | TOOL_RESULT_TYPES:
   691→            continue
   692→        if block_type in TEXT_BLOCK_TYPES:
   693→            text = str(block.get("text") or "").strip()
   694→            if text:
   695→                parts.append(text)
   696→            continue
   697→        fallback_text = extract_simple_text(block)
   698→        if fallback_text:
   699→            parts.append(fallback_text)
   700→    return join_paragraphs(parts)

   727→    if isinstance(value, list):

   738→            return [text] if text else []

   740→        for key in ("parts", "content"):

   747→def find_first_mapping_with_keys(value: Any, keys: set[str]) -> dict[str, Any] | None:
   748→    if isinstance(value, dict):
   749→        if keys & set(value.keys()):
   750→            return value
   751→        for candidate in value.values():
   752→            result = find_first_mapping_with_keys(candidate, keys)
   753→            if result is not None:
   754→                return result
   755→        return None
   756→    if isinstance(value, list):
   757→        for candidate in value:
   758→            result = find_first_mapping_with_keys(candidate, keys)
   759→            if result is not None:
   760→                return result
   761→    return None

   764→def normalize_timestamp(value: Any, fallback: str | None = None) -> str:
   765→    if isinstance(value, (int, float)):
   766→        return (
   767→            datetime.fromtimestamp(float(value), tz=timezone.utc)
   768→            .replace(microsecond=0)
   769→            .isoformat()
   770→            .replace("+00:00", "Z")
   771→        )
   772→    if isinstance(value, datetime):
   773→        normalized = value.astimezone(timezone.utc) if value.tzinfo else value

### 9. Key Design Decisions

1. Deterministic over AI-Generated: Transcript cleaning uses code, not model tokens. Raw transcripts preserved as source of truth.

2. Non-Destructive Upgrade: Existing AGENTS.md and policy files are preserved and validated, not overwritten.

3. Concurrent-Safe Global Archive: Uses SQLite WAL mode for multi-project updates without corruption.

4. Repo-Scoped Mirrors: Current project's transcripts mirrored into repo; other projects archived globally only.

5. Client Auto-Detection: Detects Codex/Claude/Gemini from path patterns without user specification.

6. Locale Auto-Selection: Detects from repo content (CJK character scoring) before system locale.

7. Zero Dependencies: All scripts use only Python standard library (json, sqlite3, pathlib, re, datetime, etc.).

8. Safe Git Defaults: Isolated PR flow, never bypasses repo rules, raw transcripts unapproved by default.
