   293→export function parseMessagesFromMarkdown(content: string): RenderedMessage[] {
   294→  const messages: RenderedMessage[] = []
   295→
   296→  // Find ## Messages section
   297→  const msgSection = content.indexOf('\n## Messages')
   298→  if (msgSection === -1) return []
   299→
   300→  const body = content.slice(msgSection + 12) // after "## Messages\n"
   301→
   302→  // Split on ### N. role headers

   304→  let match: RegExpExecArray | null
   305→  const starts: Array<{ index: number; role: string }> = []
   306→
   307→  while ((match = msgRe.exec(body)) !== null) {
   308→    starts.push({ index: match.index, role: match[1] ?? 'unknown' })
   309→  }
   310→
   311→  for (let i = 0; i < starts.length; i++) {
   312→    const start = starts[i]
   313→    if (!start) continue

   315→    const chunk = nextStart
   316→      ? body.slice(start.index, nextStart.index)
   317→      : body.slice(start.index)
   318→

   324→
   325→    for (const line of lines) {
   326→      if (!pastHeader && line.startsWith('### ')) {
   327→        pastHeader = true
   328→        continue
   329→      }
   330→      if (!pastHeader) continue

   334→        continue
   335→      }

   337→    }

   339→    // Remove leading/trailing blank lines from text

   344→  }
