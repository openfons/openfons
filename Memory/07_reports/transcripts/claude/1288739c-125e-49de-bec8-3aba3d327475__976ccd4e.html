   187→export function intervalToSeconds(interval: string): number {
   188→  const match = interval.trim().toLowerCase().match(/^(\d+)\s*(s|m|h)$/)
   189→  if (!match) {
   190→    return intervalToSeconds(DEFAULT_HEARTBEAT_INTERVAL)
   191→  }
   192→  const value = parseInt(match[1]!, 10)
   193→  const unit = match[2]!
   194→  if (value <= 0) {
   195→    return intervalToSeconds(DEFAULT_HEARTBEAT_INTERVAL)
   196→  }
   197→  const multiplier: Record<string, number> = { s: 1, m: 60, h: 3600 }
   198→  return value * (multiplier[unit] ?? 60)
   199→}

   499→function normalizeProjectEntry(project: ProjectLike, cfg: Config): ProjectEntry {
   500→  const rawPath = maybeNormalizePath(project.path)
   501→  const developmentPath = maybeNormalizePath(project.development_path) || rawPath
   502→  const memoryPath = maybeNormalizePath(project.memory_path) || rawPath || developmentPath
   503→  const canonicalPath = memoryPath || rawPath || developmentPath
   504→
   505→  if (!canonicalPath) {
   506→    throw new Error('Project entry requires at least one path.')
   507→  }
   508→
   509→  const resolvedDevelopmentPath = developmentPath || canonicalPath
   510→  const resolvedMemoryPath = memoryPath || canonicalPath
   511→  const name = nonEmptyString(project.name, basename(resolvedDevelopmentPath))
   512→  const id = createProjectId(
   513→    typeof project.id === 'string' ? project.id : '',
   514→    resolvedDevelopmentPath,
   515→    resolvedMemoryPath,
   516→  )
   517→
   518→  return {
   519→    id,
   520→    path: canonicalPath,
   521→    name,
   522→    development_path: resolvedDevelopmentPath,
   523→    memory_path: resolvedMemoryPath,
   524→    memory_branch: nonEmptyString(project.memory_branch, DEFAULT_MEMORY_BRANCH),
   525→    heartbeat_interval: isValidInterval(project.heartbeat_interval ?? '') ? project.heartbeat_interval! : cfg.heartbeat_interval,
   526→    refresh_interval: isValidInterval(project.refresh_interval ?? '') ? project.refresh_interval! : DEFAULT_REFRESH_INTERVAL,
   527→    auto_push: typeof project.auto_push === 'boolean' ? project.auto_push : cfg.auto_push,
   528→    generate_report: typeof project.generate_report === 'boolean' ? project.generate_report : cfg.generate_report,
   529→    ai_summary_model: nonEmptyString(project.ai_summary_model, cfg.ai_summary_model),
   530→    locale: nonEmptyString(project.locale, cfg.locale),
   531→    gh_pages_branch: stringOrDefault(project.gh_pages_branch, cfg.gh_pages_branch),
   532→    cname: stringOrDefault(project.cname, cfg.cname),
   533→    webhook_url: stringOrDefault(project.webhook_url, cfg.webhook_url),
   534→    report_base_url: stringOrDefault(project.report_base_url, cfg.report_base_url),
   535→    report_port: isValidPort(project.report_port) ? project.report_port : cfg.report_port,
